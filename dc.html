Automatic toll gate/ parking system using arduino

This project involves arduino, ultrasonic sensor and servo motor. the sensor detects the car and sends the signal to the motor. This motor works as a gate to the project

Code Snippets

Here's the code for the project


#include 

Servo myservo;  

int pos = 0; // Servo position
int cm = 0;  // Distance in cm

long readUltrasonicDistance(int triggerPin, int echoPin) {
  pinMode(triggerPin, OUTPUT);
  digitalWrite(triggerPin, LOW);
  delayMicroseconds(2);
  digitalWrite(triggerPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(triggerPin, LOW);
  
  pinMode(echoPin, INPUT);
  long duration = pulseIn(echoPin, HIGH, 40000); // Timeout after 40ms
  return duration;
}

void setup() {
  pinMode(6, OUTPUT); // Trigger pin
  pinMode(7, INPUT);  // Echo pin
  myservo.attach(9);  // Servo connected to pin 9
  Serial.begin(9600);
}

void loop() {
  long duration = readUltrasonicDistance(6, 7);
  cm = duration * 0.034 / 2; // Convert duration to cm
  
  // Debugging: Print the measured distance
  Serial.print("Distance: ");
  Serial.print(cm);
  Serial.println(" cm");

  // Only move the servo if an object is detected within 30 cm
  if (cm > 0 && cm < 30) {
    Serial.println("Object detected!");

    // Move the servo from 0째 to 120째
    for (pos = 0; pos <= 120; pos += 1) {
      myservo.write(pos);
      delay(15);
    }
    delay(500); // Pause for half a second

    // Move the servo back from 120째 to 0째
    for (pos = 120; pos >= 0; pos -= 1) {
      myservo.write(pos);
      delay(15);
    }
    delay(500); // Pause for half a second
  } else {
    Serial.println("No object detected.");
  }

  delay(1000); // Small delay for stable readings
}
