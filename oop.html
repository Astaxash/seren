Billing system using Java

The "customer Billing System" excels in efficiently managing billing information through dynamic structures. With an interactive console interface, it ensures a user-friendly experience and supports versatile billing scenarios. The robust business logic calculates accurate totals, applies discounts, and generates detailed receipts, empowering users with informed decision-making. The system's adaptability and graceful termination feature enhance its utility, making it a sophisticated solution for diverse billing needs.

import java.util.HashMap;
import java.util.Map;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

class BillingSystem {
    private Map products = new HashMap<>();
    private Map services = new Hash Map<>();
    private Map rents = new HashMap<>();
    private List>emiPayments = new Array List<>();
    private String reminder = "";

    public void addProduct(String name, double amount) {
products.put(name, amount);
    }

    public void addService(String name, double amount) {
services.put(name, amount);
    }

    public void addRent(String description, double amount) {
rents.put(description, amount);
    }

    public void addEmiPayment(String description, double amount, int duration) {
        MapemiPayment = new HashMap<>();
emiPayment.put("description", description);
emiPayment.put("amount", amount);
emiPayment.put("duration", duration);
emiPayments.add(emiPayment);
    }

    public void setReminder(String message) {
        reminder = message;
    }

    public void generateReceipt() {
        double totalAmount = products.values().stream().mapToDouble(Double::doubleValue).sum() +
services.values().stream().mapToDouble(Double::doubleValue).sum() +
rents.values().stream().mapToDouble(Double::doubleValue).sum();

        double totalEmiAmount = emiPayments.stream().mapToDouble(payment ->
                (double) payment.get("amount")). sum();

        double discount = 0.1 * totalAmount;
        double grandTotal = totalAmount + totalEmiAmount - discount;

        // Display Final Receipt
System.out.println("--- Final Receipt ---");
System.out.println("Products:");
products.forEach((product, amount) ->System.out.println(product + ": $" + amount));

System.out.println("\nServices:");
services.forEach((service, amount) ->System.out.println(service + ": $" + amount));

System.out.println("\nRents:");
rents.forEach((rent, amount) ->System.out.println(rent + ": $" + amount));

System.out.println("\nEMI Payments:");
emiPayments.forEach(payment ->
System.out.println(payment.get("description") + " (" + payment.get("duration") +
                        " months): $" + payment.get("amount") + "/month"));

System.out.println("\nDiscount (10%): $" + discount);
System.out.println("Grand Total: $" + grandTotal);
System.out.println("--- Thank you for your business! ---");
    }
}

public class Main {
    public static void main(String[] args) {
BillingSystembillingSystem = new BillingSystem();
        Scanner scanner = new Scanner(System.in);

        while (true) {
System.out.println("\n1. Products Bill\n2. Services Bill\n3. Rents Bill\n4. EMI Payment Bill\n5. Set Reminder\n6. Generate Receipt\n7. Exit");
System.out.print("Select an option: ");
            String choice = scanner.nextLine();

            switch (choice) {
                case "1":
System.out.print("Enter product name: ");
                    String productName = scanner.nextLine();
System.out.print("Enter product amount: $");
                    double productAmount = Double.parseDouble(scanner.nextLine());
billingSystem.addProduct(productName, productAmount);
                    break;

                case "2":
System.out.print("Enter service name: ");
                    String serviceName = scanner.nextLine();
System.out.print("Enter service amount: $");
                    double serviceAmount = Double.parseDouble(scanner.nextLine());
billingSystem.addService(serviceName, serviceAmount);
                    break;

                case "3":
System.out.print("Enter rent description: ");
                    String rentDescription = scanner.nextLine();
System.out.print("Enter rent amount: $");
                    double rentAmount = Double.parseDouble(scanner.nextLine());
billingSystem.addRent(rentDescription, rentAmount);
                    break;

                case "4":
System.out.print("Enter EMI description: ");
                    String emiDescription = scanner.nextLine();
System.out.print("Enter EMI amount: $");
                    double emiAmount = Double.parseDouble(scanner.nextLine());
System.out.print("Enter EMI duration (months): ");
intemiDuration = Integer.parseInt(scanner.nextLine());
billingSystem.addEmiPayment(emiDescription, emiAmount, emiDuration);
                    break;

                case "5":
System.out.print("Enter reminder message: ");
                    String reminderMessage = scanner.nextLine();
billingSystem.setReminder(reminderMessage);
                    break;

                case "6":
billingSystem.generateReceipt();
                    break;

                case "7":
scanner.close();
System.exit(0);

                default:
System.out.println("Invalid choice. Please enter a number between 1 and 7.");
                    break;
            }
        }
    }
}
